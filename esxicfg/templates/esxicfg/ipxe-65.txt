#!ipxe

iseq ${platform} efi && goto is_efi || goto is_bios

:is_bios
echo ESXi Installer BIOS mode
kernel http://installers.packet.cloud/{{ esxi_version }}/mboot.c32 -c http://{{ SITE_FQDN }}/esxibootcfg/{{ node_id }} gdbPort=none logPort=none tty2port=com2 BOOTIF=01-${net0/mac}
boot

:is_efi
echo ESXi Installer (U)EFI mode
kernel http://installers.packet.cloud/{{ esxi_version }}/EFI/BOOT/BOOTX64.EFI -c http://{{ SITE_FQDN }}/esxibootcfg/{{ node_id }} gdbPort=none logPort=none tty2port=com2 BOOTIF=01-${net0/mac}
boot
